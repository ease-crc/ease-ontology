def printdict(class_dict):
	for key,val in class_dict.items() :
		print(r'\textbf{{{}}}'.format(key))
		for j,k in val.items():
			print(r'\textit{{{}}}'.format(j),end=" ")
			print(r'{\textbullet}{}')
			if isinstance(k,list):
				print(', '.join(k),end=" ")
			else:
				print(k,end=" ")
		print(r"\linebreak")
\end{pycode}
